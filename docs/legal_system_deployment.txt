# Legal Case Management System - Deployment Configuration
# Production-ready Kubernetes deployment with Your County compliance

apiVersion: v1
kind: Namespace
metadata:
  name: legal-case-management
  labels:
    environment: production
    compliance: cook-county-illinois

---
# Database Configuration
apiVersion: postgresql.cnpg.io/v1
kind: Cluster
metadata:
  name: legal-postgres
  namespace: legal-case-management
spec:
  instances: 3
  primaryUpdateStrategy: unsupervised
  
  postgresql:
    parameters:
      max_connections: "200"
      shared_buffers: "256MB"
      effective_cache_size: "1GB"
      work_mem: "4MB"
      maintenance_work_mem: "64MB"
      checkpoint_completion_target: "0.9"
      wal_buffers: "16MB"
      default_statistics_target: "100"
      random_page_cost: "1.1"
      effective_io_concurrency: "200"
    
  bootstrap:
    initdb:
      database: legal_case_management
      owner: legal_user
      secret:
        name: legal-postgres-credentials
      
  storage:
    size: 100Gi
    storageClass: fast-ssd
    
  monitoring:
    enabled: true
    
  backup:
    retention: "30d"
    barmanObjectStore:
      destinationPath: s3://chitty-legal-backups/postgres
      s3Credentials:
        accessKeyId:
          name: s3-credentials
          key: ACCESS_KEY_ID
        secretAccessKey:
          name: s3-credentials
          key: SECRET_ACCESS_KEY
      wal:
        retention: "7d"
      data:
        retention: "30d"

---
# Redis for caching and session management
apiVersion: apps/v1
kind: Deployment
metadata:
  name: legal-redis
  namespace: legal-case-management
spec:
  replicas: 2
  selector:
    matchLabels:
      app: legal-redis
  template:
    metadata:
      labels:
        app: legal-redis
    spec:
      containers:
      - name: redis
        image: redis:7-alpine
        ports:
        - containerPort: 6379
        resources:
          requests:
            memory: "256Mi"
            cpu: "100m"
          limits:
            memory: "512Mi"
            cpu: "200m"
        volumeMounts:
        - name: redis-config
          mountPath: /usr/local/etc/redis/redis.conf
          subPath: redis.conf
      volumes:
      - name: redis-config
        configMap:
          name: redis-config

---
# Main API Service
apiVersion: apps/v1
kind: Deployment
metadata:
  name: legal-api
  namespace: legal-case-management
spec:
  replicas: 3
  selector:
    matchLabels:
      app: legal-api
  template:
    metadata:
      labels:
        app: legal-api
    spec:
      containers:
      - name: legal-api
        image: chitty-stacks/legal-api:latest
        ports:
        - containerPort: 3000
        env:
        - name: DATABASE_URL
          valueFrom:
            secretKeyRef:
              name: legal-postgres-credentials
              key: uri
        - name: REDIS_URL
          value: "redis://legal-redis:6379"
        - name: OPENAI_API_KEY
          valueFrom:
            secretKeyRef:
              name: openai-credentials
              key: api-key
        - name: VECTOR_STORE_PRIMARY
          value: "your-vector-store-id"
        - name: VECTOR_STORE_LLC
          value: "your-vector-store-id"
        - name: CLERK_SECRET_KEY
          valueFrom:
            secretKeyRef:
              name: clerk-credentials
              key: secret-key
        - name: COOK_COUNTY_COMPLIANCE_MODE
          value: "true"
        - name: EVIDENCE_RETENTION_YEARS
          value: "7"
        resources:
          requests:
            memory: "512Mi"
            cpu: "250m"
          limits:
            memory: "1Gi"
            cpu: "500m"
        livenessProbe:
          httpGet:
            path: /health
            port: 3000
          initialDelaySeconds: 30
          periodSeconds: 10
        readinessProbe:
          httpGet:
            path: /ready
            port: 3000
          initialDelaySeconds: 5
          periodSeconds: 5

# Deadline Management CronJob
apiVersion: batch/v1
kind: CronJob
metadata:
  name: deadline-checker
  namespace: legal-case-management
spec:
  schedule: "0 */6 * * *" # Every 6 hours
  jobTemplate:
    spec:
      template:
        spec:
          containers:
          - name: deadline-checker
            image: chitty-stacks/deadline-checker:latest

            - name: DATABASE_URL
              valueFrom:
                secretKeyRef:
                  name: legal-postgres-credentials
                  key: uri

                memory: "256Mi"
                cpu: "100m"

                memory: "512Mi"

# Network Policy for Security
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: legal-api-network-policy
  namespace: legal-case-management
spec:
  podSelector:
    matchLabels:
      app: legal-api
  policyTypes:
  - Ingress
  - Egress
  ingress:
  - from:
    - namespaceSelector:
        matchLabels:
          name: ingress-nginx
  egress:
  - to:
    - namespaceSelector:
        matchLabels:
          name: kube-system
  - to:
    - podSelector:
        matchLabels:
          app: legal-postgres
    ports:
    - protocol: TCP
      port: 5432
  - to:
    - podSelector:
        matchLabels:
          app: legal-redis
    ports:
    - protocol: TCP
      port: 6379
